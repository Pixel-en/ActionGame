void Bard::Update()
{

	{

		UpdateIdol();

	case Enemy::ATTACK:
		UpdateAttack();
		break;
	case Enemy::MOVE:
		UpdateMove();
		break;
	case Enemy::DAMEGE:
		UpdateDamege();
		break;
	case Enemy::DEATH:

		break;
	}
}
